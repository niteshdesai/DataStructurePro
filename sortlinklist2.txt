#include<stdio.h>

#include<stdlib.h>

struct node
{
	int data;
	struct node *next;
};


struct node* insert(struct node *head,int ele)
{
	struct node *n,*prev,*curr;

	n=(struct node*)malloc(sizeof(struct node));

    n->data=ele;
	prev=NULL;
	curr=head;
     
	if(head==NULL)
	{
	   
		n->next=NULL;
		head=n;

	}
	else
	{

		while( curr!=NULL && ele>curr->data)
		{
			prev=curr;
			curr=curr->next;
		}
		
        
		if(head==curr)
		{
           n->next=head;
		   head=n;
		}

		else
		{
	
			prev->next=n;
			n->next=curr;
		}
	}

  return head;
}

void display(struct node *head)
{
	struct node *temp;
	temp=head;
    

    while(temp!=NULL)
	{
		printf("%d ",temp->data);
		temp=temp->next;
	}
}

}
